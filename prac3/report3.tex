\section{除算を行うプログラム}
\normalsize
プログラムのソースと実行結果を以下に示す。
\footnotesize
\begin{itembox}[l]{ソース}
\begin{verbatim}
/* 8.c */
#include <stdio.h>

int main()
{
  int x,y,i;

  printf("x : ");
  scanf("%d",&x);

  printf("y : ");
  scanf("%d",&y);

  if (y == 0){                  /* 0 で割るを禁止 */
    printf("error\n");
    return -1;
  }

  printf("%d",x/y);             /* 整数部分の商を表示 */

  if(x%y!=0){
    printf(".");
    for(i=0;i<20;i++){
      x = x % y * 10;             /* 余りを出し10倍する */
      if(x == 0)                  /* 余りが 0 であったらループを抜ける */
        break;
      printf("%d",x / y );
    }
  }
  printf("\n");

  return 0;
}
\end{verbatim}
\end{itembox}

/* 9.c */
#include <stdio.h>

int main()
{
  int t, f, turu, kame, i;

  printf("Total : ");
  scanf("%d",&t);

  printf("Number of foot : ");
  scanf("%d", &f);

  turu = t;                     /* すべて鶴とみなす */
  kame = 0;

  while(turu * 2 + kame * 4 != f){
    turu--;
    kame++;
    if(turu<0){
      printf("No answer\n");
      return -1;
    }
  }

  printf("Turu : %d, Kame : %d\n", turu, kame);

  return 0;
}
